\documentclass[tikz,border=1.35mm]{standalone}

\definecolor{fvmadaptedge}{HTML}{8A7E7E}
\usetikzlibrary{arrows.meta}

\definecolor{splitred}{HTML}{AF2525}
\definecolor{centroidgreen}{HTML}{0E8814}

% Wireframe exploded view of the vertex cells produced by pyramid refinement.
% The five separated vertex cells duplicate shared edge-midpoint, face-centroid,
% and volume-centroid vertices so each child can be read independently.
\providecommand{\fvmadaptYawAngle}{8}
\providecommand{\fvmadaptExplodeAmount}{1}
\providecommand{\fvmadaptExplodeScale}{1.3}
\pgfmathsetmacro{\yawcos}{cos(\fvmadaptYawAngle)}
\pgfmathsetmacro{\yawsin}{sin(\fvmadaptYawAngle)}
\pgfmathsetmacro{\xbasisx}{20.3*\yawcos + 15.3*\yawsin}
\pgfmathsetmacro{\xbasisy}{-8.5*\yawcos + 11.3*\yawsin}
\pgfmathsetmacro{\ybasisx}{-20.3*\yawsin + 15.3*\yawcos}
\pgfmathsetmacro{\ybasisy}{8.5*\yawsin + 11.3*\yawcos}
\pgfmathsetmacro{\fvmadaptEtaProjectionScale}{0.75}
\pgfmathsetmacro{\ybasisxscaled}{\fvmadaptEtaProjectionScale*\ybasisx}
\pgfmathsetmacro{\ybasisyscaled}{\fvmadaptEtaProjectionScale*\ybasisy}
\pgfmathsetmacro{\explodeXi}{0.42*\fvmadaptExplodeScale*\fvmadaptExplodeAmount}
\pgfmathsetmacro{\explodeEta}{0.32*\fvmadaptExplodeScale*\fvmadaptExplodeAmount}
\pgfmathsetmacro{\explodeZeta}{0.62*\fvmadaptExplodeScale*\fvmadaptExplodeAmount}

\begin{document}
\begin{tikzpicture}[
  x={(\xbasisx mm,\xbasisy mm)},
  y={(\ybasisxscaled mm,\ybasisyscaled mm)},
  z={(0mm,20mm)},
  line join=round,
  line cap=round,
  outer/.style={draw=fvmadaptedge,line width=0.54mm},
  split/.style={draw=splitred,line width=0.42mm},
  centroidedge/.style={draw=centroidgreen,line width=0.48mm},
  axis/.style={draw=fvmadaptedge!60,line width=0.32mm,-{Latex[length=2.5mm,width=1.8mm]}}
]
  % Fixed canvas for static and animated renders. The yaw sweep changes the
  % projected x/y basis, so a stable canvas avoids per-frame cropping.
  \path[use as bounding box]
    (canvas cs:x=-58mm,y=-60mm) rectangle (canvas cs:x=68mm,y=66mm);

  \newcommand{\drawbasevertexcell}[8]{%
    \draw[outer] (#1) -- (#2);
    \draw[outer] (#1) -- (#3);
    \draw[outer] (#1) -- (#4);
    \draw[split] (#2) -- (#5);
    \draw[split] (#3) -- (#5);
    \draw[split] (#2) -- (#6);
    \draw[split] (#4) -- (#6);
    \draw[split] (#3) -- (#7);
    \draw[split] (#4) -- (#7);
    \draw[centroidedge] (#5) -- (#8);
    \draw[centroidedge] (#6) -- (#8);
    \draw[centroidedge] (#7) -- (#8);
    \fill[fvmadaptedge] (#1) circle[radius=0.58mm];
    \foreach \p in {#2,#3,#4,#5,#6,#7} {
      \fill[splitred] (\p) circle[radius=0.50mm];
    }
    \fill[centroidgreen] (#8) circle[radius=0.66mm];
  }

  \newcommand{\drawapexcell}[9]{%
    \draw[outer] (#1) -- (#2);
    \draw[outer] (#1) -- (#3);
    \draw[outer] (#1) -- (#4);
    \draw[outer] (#1) -- (#5);
    \draw[split] (#2) -- (#6);
    \draw[split] (#3) -- (#6);
    \draw[split] (#3) -- (#7);
    \draw[split] (#4) -- (#7);
    \draw[split] (#4) -- (#8);
    \draw[split] (#5) -- (#8);
    \draw[split] (#5) -- (#9);
    \draw[split] (#2) -- (#9);
    \foreach \p in {#6,#7,#8,#9} {
      \draw[centroidedge] (\p) -- (CV);
    }
    \fill[fvmadaptedge] (#1) circle[radius=0.58mm];
    \foreach \p in {#2,#3,#4,#5,#6,#7,#8,#9} {
      \fill[splitred] (\p) circle[radius=0.50mm];
    }
    \fill[centroidgreen] (CV) circle[radius=0.66mm];
  }

  % Compact orientation axes near the fixed xi/eta/zeta-minimum child.
  \coordinate (Oaxis) at (-1.28,-0.55,-0.84);
  \draw[axis] (Oaxis) -- (-0.32,-0.55,-0.84) node[below right,font=\normalsize,text=fvmadaptedge] {$\xi$};
  \draw[axis] (Oaxis) -- (-1.28,0.39,-0.84) node[above left,font=\normalsize,text=fvmadaptedge] {$\eta$};
  \draw[axis] (Oaxis) -- (-1.28,-0.55,0.12) node[above,font=\normalsize,text=fvmadaptedge] {$\zeta$};

  % Explosion heuristic: keep the base xi/eta minimum child fixed, slide the
  % other base children outward in xi/eta, and lift the apex child in zeta.

  % Child near base vertex A: edges AB, AD, and AE.
  \begin{scope}[shift={(0,0,0)}]
    \coordinate (O) at (0,0,0);
    \coordinate (X) at (0.725,0,0);
    \coordinate (Y) at (0,0.5,0);
    \coordinate (Z) at (0.3625,0.25,0.5);
    \coordinate (XY) at (0.725,0.5,0);
    \coordinate (XZ) at (0.725,0.1667,0.3333);
    \coordinate (YZ) at (0.2417,0.5,0.3333);
    \coordinate (CV) at (0.725,0.5,0.25);
    \drawbasevertexcell{O}{X}{Y}{Z}{XY}{XZ}{YZ}{CV}
  \end{scope}

  % Child near base vertex B: edges BA, BC, and BE.
  \begin{scope}[shift={(\explodeXi,0,0)}]
    \coordinate (O) at (1.45,0,0);
    \coordinate (X) at (0.725,0,0);
    \coordinate (Y) at (1.45,0.5,0);
    \coordinate (Z) at (1.0875,0.25,0.5);
    \coordinate (XY) at (0.725,0.5,0);
    \coordinate (XZ) at (0.725,0.1667,0.3333);
    \coordinate (YZ) at (1.2083,0.5,0.3333);
    \coordinate (CV) at (0.725,0.5,0.25);
    \drawbasevertexcell{O}{X}{Y}{Z}{XY}{XZ}{YZ}{CV}
  \end{scope}

  % Child near base vertex C: edges CB, CD, and CE.
  \begin{scope}[shift={(\explodeXi,\explodeEta,0)}]
    \coordinate (O) at (1.45,1,0);
    \coordinate (X) at (1.45,0.5,0);
    \coordinate (Y) at (0.725,1,0);
    \coordinate (Z) at (1.0875,0.75,0.5);
    \coordinate (XY) at (0.725,0.5,0);
    \coordinate (XZ) at (1.2083,0.5,0.3333);
    \coordinate (YZ) at (0.725,0.8333,0.3333);
    \coordinate (CV) at (0.725,0.5,0.25);
    \drawbasevertexcell{O}{X}{Y}{Z}{XY}{XZ}{YZ}{CV}
  \end{scope}

  % Child near base vertex D: edges DC, DA, and DE.
  \begin{scope}[shift={(0,\explodeEta,0)}]
    \coordinate (O) at (0,1,0);
    \coordinate (X) at (0.725,1,0);
    \coordinate (Y) at (0,0.5,0);
    \coordinate (Z) at (0.3625,0.75,0.5);
    \coordinate (XY) at (0.725,0.5,0);
    \coordinate (XZ) at (0.725,0.8333,0.3333);
    \coordinate (YZ) at (0.2417,0.5,0.3333);
    \coordinate (CV) at (0.725,0.5,0.25);
    \drawbasevertexcell{O}{X}{Y}{Z}{XY}{XZ}{YZ}{CV}
  \end{scope}

  % Child near apex E: edges EA, EB, EC, and ED.
  \begin{scope}[shift={(0.5*\explodeXi,0.5*\explodeEta,\explodeZeta)}]
    \coordinate (O) at (0.725,0.5,1);
    \coordinate (AEdge) at (0.3625,0.25,0.5);
    \coordinate (BEdge) at (1.0875,0.25,0.5);
    \coordinate (CEdge) at (1.0875,0.75,0.5);
    \coordinate (DEdge) at (0.3625,0.75,0.5);
    \coordinate (FABE) at (0.725,0.1667,0.3333);
    \coordinate (FBCE) at (1.2083,0.5,0.3333);
    \coordinate (FCDE) at (0.725,0.8333,0.3333);
    \coordinate (FDAE) at (0.2417,0.5,0.3333);
    \coordinate (CV) at (0.725,0.5,0.25);
    \drawapexcell{O}{AEdge}{BEdge}{CEdge}{DEdge}{FABE}{FBCE}{FCDE}{FDAE}
  \end{scope}
\end{tikzpicture}
\end{document}
